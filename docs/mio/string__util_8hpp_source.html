88 template <typename CharT, typename S>
89 struct is_c_str_helper
90 {
91 using Decayed = typename std::decay<S>::type;
92 using Base = typename std::remove_cv<typename std::remove_pointer<Decayed>::type>::type;
93 static constexpr bool value = std::is_same<Base, CharT>::value;
94 };

102 template <typename S>

104 {
105 static constexpr bool value = is_c_str_helper<char, S>::value
106#ifdef _WIN32

108#endif
109 ;

112 template <typename String, typename = decltype(std::declval<String>().data()),
113 typename = typename std::enable_if<!is_c_str_or_c_wstr<String>::value>::type>
