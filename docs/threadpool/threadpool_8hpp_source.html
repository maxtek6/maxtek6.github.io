64 threadpool(size_t threads = std::thread::hardware_concurrency())
65 {
66 if (threads == 0)
67 {
68 throw std::runtime_error("failed to construct threadpool with zero threads");
69 }
70
71 _active = true;
72
73 _workers.reserve(threads);
74
75 while (_workers.size() < _workers.capacity())
76 {
77 _workers.push_back(std::thread([&]()
78 {
79 std::function<void()> task;
80 while (pop_task(task))
81 {
82 task();
83 } }));
84 }
85 }
89 ~threadpool()
90 {
91 if (_active)
92 {
93 shutdown();
94 }
95 }

106 template <class F, class... Args>

108 {

114 result = packaged_task->get_future();
115
116 work = [packaged_task]()

121 push_task(std::move(work));

154 void push_task(std::function<void()> &&task)
155 {
156 std::unique_lock<std::mutex> lock(_mutex);
157 if (!_active)
158 {
159 throw std::runtime_error("failed to submit to inactive threadpool");
160 }
161 _tasks.push(std::move(task));
162 lock.unlock();
163 _condition.notify_one();
164 }
165
166 bool pop_task(std::function<void()> &task)
167 {
168 std::unique_lock<std::mutex> lock(_mutex);
169 bool result(false);
170 _condition.wait(
171 lock,
172 [&]()
173 {
174 return (!_active || !_tasks.empty());
175 });
176 if (_active)
177 {
178 task = _tasks.front();
179 _tasks.pop();
180 result = true;
181 }
182 return result;
183 }
